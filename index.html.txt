    <script>
        // 分頁切換
        function switchPage(pageId) {
            const p1 = document.getElementById('page1');
            const p2 = document.getElementById('page2');
            const b1 = document.getElementById('btn-page1');
            const b2 = document.getElementById('btn-page2');

            if (pageId === 'page1') {
                p1.classList.remove('hidden');
                p2.classList.add('hidden');
                b1.className = b1.className.replace('bg-blue-800 text-white', 'bg-white text-blue-700');
                b2.className = b2.className.replace('bg-white text-blue-700', 'bg-blue-800 text-white');
            } else {
                p2.classList.remove('hidden');
                p1.classList.add('hidden');
                b2.className = b2.className.replace('bg-blue-800 text-white', 'bg-white text-blue-700');
                b1.className = b1.className.replace('bg-white text-blue-700', 'bg-blue-800 text-white');
            }
        }

        // 上傳照片
        function handleImageUpload(event) {
            const files = event.target.files;
            const container = document.getElementById('photoContainer');
            if (container.querySelector('.italic')) container.innerHTML = '';

            Array.from(files).forEach(file => {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const card = document.createElement('div');
                    card.className = 'photo-card relative shadow-md';
                    card.innerHTML = `
                        <button onclick="this.parentElement.remove()" class="no-print absolute -top-3 -right-3 bg-red-600 text-white w-8 h-8 rounded-full border-2 border-white shadow-lg">✕</button>
                        <img src="${e.target.result}" class="photo-preview">
                        <textarea placeholder="點此輸入說明..." class="text-sm border-b border-gray-300 p-2 focus:bg-yellow-50" rows="2"></textarea>
                    `;
                    container.appendChild(card);
                };
                reader.readAsDataURL(file);
            });
        }

        // 初始化日期
        window.onload = function() {
            const d = new Date();
            const year = d.getFullYear() - 1911;
            const month = String(d.getMonth() + 1).padStart(2, '0');
            const day = String(d.getDate()).padStart(2, '0');
            document.getElementById('currentDate').value = `${year} 年 ${month} 月 ${day} 日`;

            window.onbeforeprint = () => {
                document.getElementById('page1').classList.remove('hidden');
                document.getElementById('page2').classList.remove('hidden');
            };
            window.onafterprint = () => {
                const isP1 = document.getElementById('btn-page1').classList.contains('bg-white');
                if (isP1) document.getElementById('page2').classList.add('hidden');
                else document.getElementById('page1').classList.add('hidden');
            };
        }
    </script>
